عنوان المهمة:
WORKFLOW PHASE 5-A — Immutable Execution Steps Snapshot

المهمة الحالية:
منع:
definition drift corruption

داخل:
approval resume lifecycle

الهدف:
كل execution تستخدم:
frozen workflow steps

من لحظة trigger
حتى completion

حتى لو:
workflow definition تغيّرت لاحقًا.

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1 — SCHEMA CHANGE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

أضف إلى:
workflow_executions

الحقول التالية:

steps_snapshot JSONB NULL
workflow_version INTEGER NULL

المطلوب:
- backward compatible
- existing executions تبقى سليمة
- no destructive migration

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
2 — ENGINE SNAPSHOT CAPTURE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

داخل:
engine.ts

عند إنشاء execution جديدة:

احفظ:
steps_snapshot = workflow.steps
workflow_version = workflow.version (إذا موجود)

مهم:
- deep clone قبل التخزين
- no shared references
- snapshot immutable by convention

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
3 — RESUME EXECUTION SOURCE OF TRUTH
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

داخل:
resumeExecution()

غيّر behavior بالكامل:

بدل:
قراءة الخطوات من:
workflow_definitions.steps

اجعل:
source of truth =
execution.steps_snapshot

المطلوب:
A)
إذا:
steps_snapshot موجودة
→ استخدمها فقط

B)
إذا:
steps_snapshot = NULL
(legacy executions)
→ fallback إلى behavior الحالي
مع structured warning log:
action='legacy_resume_live_definition'

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
4 — REMOVE LIVE DEFINITION DEPENDENCY
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

داخل resume flow:

لا تعتمد على:
workflow_definitions.steps

إلا legacy fallback فقط.

المطلوب:
أي execution جديدة بعد P5-A
لا يمكن أن تتأثر بأي تعديل لاحق
على workflow definition.
