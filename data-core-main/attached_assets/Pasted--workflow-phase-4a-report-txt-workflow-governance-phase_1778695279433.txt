عنوان المهمة:
WORKFLOW PHASE 4-B — Lazy TTL & Stuck Execution Protection

المهمة الحالية:
إضافة execution timeout protection
للـ workflow runtime
بدون إدخال:
- schedulers
- queues
- cron systems
- distributed timers

الهدف:
منع:
- stuck executions
- zombie workflows
- endless suspended executions

بطريقة:
single-process pragmatic architecture

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1 — SCHEMA CHANGE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

أضف إلى:
workflow_executions

الحقل:
timeout_at TIMESTAMP NULL

ثم:
status enum/value:
أضف:
timed_out

المطلوب:
- migration واضحة
- backward compatible
- existing rows تبقى سليمة

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
2 — DEFAULT EXECUTION TTL
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

داخل:
engine.ts

عند إنشاء execution:

احسب:
timeoutAt = now + DEFAULT_EXECUTION_TTL_HOURS

الافتراضي:
24 ساعة

المطلوب:
- configurable constant
- documented clearly
- persisted في DB

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
3 — INTER-STEP TTL CHECK
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

داخل:
executor.ts

عند:
inter-step boundary

(نفس المنطقة التي أصبحت نظيفة بعد P4-A)

أضف:

إذا:
now > timeout_at

إذًا:
- execution.status = "timed_out"
- write structured warning log
- stop execution safely
- no throw
- no partial next-step execution

المطلوب:
- cooperative timeout
- current running step finishes first
- timeout only checked BETWEEN steps

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
4 — TIMEOUT DIAGNOSTICS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

أنشئ endpoint:

GET /api/workflows/executions/stuck

المقصود:
Executions:
- status = running OR waiting_approval
- timeout_at < now

المطلوب:
إرجاع:
- execution id
- workflow id/name
- workspaceId
- trigger event
- startedAt
- timeoutAt
- computed overdue duration
- currentStepIndex

filters:
- workspaceId
- workflowId
- overdueMinutes

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
5 — ADMIN TIMEOUT ACTION
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

أنشئ endpoint:

POST /api/workflows/executions/:id/timeout

super_admin only.

المطلوب:
- force status → timed_out
- structured audit log
- no hard delete
- no execution mutation beyond status

إذا execution already terminal:
- return safe validation error

7 — SAFETY RULES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

per execution.
