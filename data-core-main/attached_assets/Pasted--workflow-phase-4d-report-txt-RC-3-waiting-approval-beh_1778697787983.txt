WORKFLOW PHASE 4-E — Approval Resume & Safe Executor Re-entry

المهمة الحالية:
تحويل:
waiting_approval

من:
dead-end paused state

safe resumable execution state

- current approval behavior

الهدف:
تمكين:
approval → workflow resume

safe cooperative re-entry model

2 — DESIGN SAFE RESUME MODEL
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

صمّم ونفّذ:
resumeExecution(executionId)

كنقطة دخول رسمية وآمنة لإعادة تشغيل executor.

المطلوب:
- execution يعيد الدخول من next step الصحيح
- no duplicate previous step execution
- no repeated side effects
- no replay of completed steps

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
3 — APPROVAL RESUME TRANSITION
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

أضف guarded transition جديد:

waiting_approval → running

لكن فقط إذا:
- approval resolved
- status ما زال waiting_approval
- execution ليست terminal
- cancel_requested = false

المطلوب:
guarded UPDATE + .returning()

نفس P4-D model.

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
4 — RESUME ENTRY SAFETY
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

داخل:
resumeExecution()

أضف protections ضد:
- duplicate resume requests
- concurrent approval actions
- resume-after-timeout
- resume-after-cancel
- resume-after-completion

المطلوب:
إذا .returning() فارغة:
- structured warn log
- safe return
- no throw

5 — STEP RESUME POSITION
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

مهم جدًا:

إذا:
step i = approval step

فبعد approval:
لا تعيد تنفيذ:

بل:
ابدأ من:
step i + 1

المطلوب:
- no duplicate notifications
- no duplicate assignments
- no duplicate side effects
